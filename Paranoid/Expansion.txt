			'oldbarney_body_014.png',

			'oldbarney_body_016.png',

			'oldbarney_body_021.png',
			'oldbarney_body_022.png',
			'oldbarney_body_023.png',
			'oldbarney_body_024.png'
		};
		static const name modelskin2[]=
		{
			'oldbarney_gun_000.png',
			'oldbarney_gun_001.png',
			'oldbarney_body_004.png',
			'oldbarney_body_019.png',
			'oldbarney_body_018.png',
			'oldbarney_body_014.png'
		};

		for(int i = 0; i <= 24; i++)
			A_ChangeModel("", 0, "Models/Expansion", "oldbarney_body.iqm", i, "Models/Expansion", modelskin1[i], CMDL_USESURFACESKIN);
		for(int i = 0; i <= 5; i++)
			A_ChangeModel("", 1, "Models/Expansion", "oldbarney_gun.iqm", i, "Models/Expansion", modelskin2[i], CMDL_USESURFACESKIN, 0);
	}
	Default
	{
		SeeSound "quiver/monsters/barney/idle";
		PainSound "quiver/monsters/barney/pain";
		DeathSound "quiver/monsters/barney/death";
		-FRIENDLY
		+COUNTKILL
		Species "Quiver";
		DropItem "HLPistolAmmo";
	}
	States
	{
	Spawn:
		HLI1 ABCDEFGHIJKLMNOPQRSTU 2 A_Look;
		Loop;
	See:
		HLR1 ABCDEFGHIJKLMNOPQRS 1 A_HLChase;
		HLR1 A 0 A_Chatter;
		Loop;
	Missile:
		HLA1 A 2 A_FaceTarget;
		HLA1 BCD 2;
		HLA1 E 2 A_FaceTarget;
		HLA1 FG 2;
	Pistol:
		HLA1 H 3 A_FaceTarget;
		HLA1 I 3;
		HLA1 J 3 Light("ImShootingAGun") A_BarneyGunshot;
		HLA1 KLM 3;
		TNT1 A 0 A_MonsterRefire(0,"See");
		Loop;
	Pain:
		HLP1 A 2 A_Pain;
		HLP1 BCD 2;
		Goto See;
	Death:
		HLD1 A 2 A_ScreamAndUnblock;
		HLD1 BCDEFG 2;
		HLD1 H 2 A_PlaySound("halflife/weapons/bodydrop");
		HLD1 IJKLMNOPQRSTUVWXYZ[\] 2;
		HLD2 ABCDEFGHIJKLMNOP 2;
		HLD2 Q -1;
		Stop;
	}
}
